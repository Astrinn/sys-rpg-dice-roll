using System;

class Program
{
    static void Main(string[] args)
    {
        // Criando um objeto para gerar números aleatórios
        Random random = new Random();

        // Loop principal para permitir múltiplas rolagens
        while (true)
        {
            Console.WriteLine("Pressione qualquer tecla para rolar os dados ou 'q' para sair.");
            var key = Console.ReadKey(true);

            // Verifica se o usuário deseja sair
            if (key.KeyChar == 'q')
                break;

            // Rolar os dados
            int[] diceRolls = RollDice(random, 6);

            // Exibir os resultados
            Console.WriteLine("Resultado do lançamento:");
            foreach (int roll in diceRolls)
            {
                Console.Write(roll + " ");
            }
            Console.WriteLine("\n");
        }
    }

    // Função para rolar os dados
    static int[] RollDice(Random random, int numberOfDice)
    {
        int[] rolls = new int[numberOfDice];

        // Rolar cada dado
        for (int i = 0; i < numberOfDice; i++)
        {
            // Gerar um número aleatório entre 1 e 6 (inclusive)
            rolls[i] = random.Next(1, 7);
        }

        return rolls;
    }
}
